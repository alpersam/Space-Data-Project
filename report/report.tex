\section{Analysis Workflow}
\label{sec:workflow}

% TODO: Walk through the notebook structure at a high level.
% Show that you understand what each step does — not just that it runs.
% A numbered list or small diagram works well here.

\begin{enumerate}
  \item \textbf{Loading \& masking} — read JPL mascon NetCDF, apply
        land mask and study-region bounding box.
  \item \textbf{Area-weighted TWS series} — compute spatially aggregated
        anomaly; apply gain factors.
  \item \textbf{Gap interpolation} — fill 2017–2018 gap; flag months.
  \item \textbf{EOF decomposition} — extract PC1 as drought index.
  \item \textbf{ERA5 P$-$ET} — download or synthesise net water balance.
  \item \textbf{MODIS NDVI} — pull cropland-masked monthly NDVI via GEE.
  \item \textbf{Futures ingestion} — download \texttt{ZS=F}; compute
        log-returns; ADF test.
  \item \textbf{Cross-correlation} — bootstrap CI; identify best lag.
  \item \textbf{Granger causality} — test at lags 1–6 months; report
        F-statistic \& p-value.
  \item \textbf{Event study} — overlay TWS with 2005, 2010, 2015, 2021
        drought years.
  \item \textbf{Backtest} — illustrative long signal when
        TWS $< -1\sigma$, shifted by best lag.
\end{enumerate}
